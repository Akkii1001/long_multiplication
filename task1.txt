#include <iostream>
#include <string.h>
using namespace std;
int main()
{
    int a[100],b[100],ans[200]={0},i,j,tmp;
    char s1[100],s2[100];
    cin>>s1;
    cin>>s2;
    for(i = strlen(s1)-1,j=0;i>=0;i--,j++)
    {
        a[j] = s1[i]-'0';
    }
    for(i = strlen(s2)-1,j=0;i>=0;i--,j++)
    {
        b[j] = s2[i]-'0';
    }
    for(i = 0;i < strlen(s2);i++)
    {
        for(j = 0;j < strlen(s1);j++)
        {
            ans[i+j] += b[i]*a[j];
        }
    }
    for(i = 0;i < strlen(s1)+strlen(s2);i++)
    {
        tmp = ans[i]/10;
        ans[i] = ans[i]%10;
        ans[i+1] = ans[i+1] + tmp;
    }
    for(i = strlen(s1)+strlen(s2); i>= 0;i--)
    {
        if(ans[i] > 0)
            break;
    }
    for(;i >= 0;i--)
    {
        cout<<ans[i];
    }
    return 0;
}
